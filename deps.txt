
# Dependencies for Sway, Waybar, and Custom Scripts

curl                  # For remote install and stream pull
git                   # Version control system
sway                  # Wayland compositor (window manager)
waybar                # Status bar for Sway
wofi                  # Application launcher for Wayland
mako                  # Notification daemon (lightweight)
# dunst               # Alternative notification daemon
# xfce4-notifyd       # Notification daemon for XFCE
libnotify             # Provides notify-send for desktop notifications
